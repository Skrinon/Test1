#include <iostream>
#include <vector>
#include <algorithm>

double calculateMean(const std::vector<double>& values) {
    double sum = 0.0;
    for (double value : values) {
        sum += value;
    }
    return values.empty() ? 0 : sum / values.size();
}

double calculateMedian(std::vector<double>& values) {
    size_t size = values.size();
    if (size == 0) {
        return 0; // Возвращаем 0 для пустого вектора
    }
    std::sort(values.begin(), values.end()); // Сортируем значения
    if (size % 2 == 0) {
        // Если четное количество элементов, возвращаем среднее двух средних значений
        return (values[size / 2 - 1] + values[size / 2]) / 2;
    }
    else {
        // Если нечетное количество, возвращаем средний элемент
        return values[size / 2];
    }
}

        double number;
        int n;

        std::cout << "Enter the number of values: ";
        std::cin >> n;

        std::cout << "Enter " << n << " numbers:\n";
        for (int i = 0; i < n; ++i) {
            std::cin >> number;
            numbers.push_back(number);
        }

        double mean = calculateMean(numbers);
        std::cout << "Mean value: " << mean << std::endl;
    }
    else if (choice == 2) {
        std::cout << "Student's Calculator functionality is under construction.\n";
    }

        for (int i = 0; i < n; ++i) {
            std::cin >> number;
            numbers.push_back(number);
        }

        double median = calculateMedian(numbers);
        std::cout << "The median is: " << median << std::endl;
    }
    else {

    }

}
